FBR Invoicing System const { useState, useEffect } = React; 
    // Main App Component
    function App() {
        const [activeTab, setActiveTab] = useState('create');
        const [invoiceData, setInvoiceData] = useState({
            invoiceType: 'Sale Invoice',
            invoiceDate: '',
            sellerNTNCNIC: '',
            sellerBusinessName: '',
            sellerProvince: '',
            sellerAddress: '',
            buyerNTNCNIC: '',
            buyerBusinessName: '',
            buyerProvince: '',
            buyerAddress: '',
            buyerRegistrationType: 'Registered',
            invoiceRefNo: '',
            scenarioId: 'SN001',
            items: [{ hsCode: '', productDescription: '', rate: '', uoM: '', quantity: 1, totalvalues: 0, valuesalesExcludingST: 0, fixedNotifiedValueOrRetailPrice: 0, salesTaxApplicable: 0, salesTaxWithheldAtSource: 0, extraTax: 0, furtherTax: 0, sroScheduleNo: '', fedPayable: 0, discount: 0, saleType: 'Goods at standard rate (default)', sroItemSerialNo: '' }]
        });
        const [provinces, setProvinces] = useState([]);
        const [docTypes, setDocTypes] = useState([]);
        const [error, setError] = useState('');

        // Mock API calls (replace with actual API endpoints)
        useEffect(() => {
            // Fetch provinces
            fetch('https://gw.fbr.gov.pk/pdi/v1/provinces', { headers: { Authorization: 'Bearer YOUR_TOKEN' } })
                .then(res => res.json())
                .then(data => setProvinces(data))
                .catch(() => setError('Failed to fetch provinces'));
            // Fetch document types
            fetch('https://gw.fbr.gov.pk/pdi/v1/doctypecode', { headers: { Authorization: 'Bearer YOUR_TOKEN' } })
                .then(res => res.json())
                .then(data => setDocTypes(data))
                .catch(() => setError('Failed to fetch document types'));
        }, []);

        const handleInputChange = (e, index = null, field = null) => {
            if (index !== null && field) {
                const newItems = [...invoiceData.items];
                newItems[index][field] = e.target.value;
                setInvoiceData({ ...invoiceData, items: newItems });
            } else {
                setInvoiceData({ ...invoiceData, [e.target.name]: e.target.value });
            }
        };

        const addItem = () => {
            setInvoiceData({
                ...invoiceData,
                items: [...invoiceData.items, { hsCode: '', productDescription: '', rate: '', uoM: '', quantity: 1, totalvalues: 0, valuesalesExcludingST: 0, fixedNotifiedValueOrRetailPrice: 0, salesTaxApplicable: 0, salesTaxWithheldAtSource: 0, extraTax: 0, furtherTax: 0, sroScheduleNo: '', fedPayable: 0, discount: 0, saleType: 'Goods at standard rate (default)', sroItemSerialNo: '' }]
            });
        };

        const handleSubmit = (endpoint) => {
            fetch(endpoint, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json', Authorization: 'Bearer YOUR_TOKEN' },
                body: JSON.stringify(invoiceData)
            })
                .then(res => res.json())
                .then(data => alert(JSON.stringify(data, null, 2)))
                .catch(() => setError('Submission failed'));
        };

        return (
            <div className="min-h-screen bg-gray-100 p-6">
                <h1 className="text-3xl font-bold text-center mb-6">FBR Invoicing System</h1>
                {error && <p className="text-red-500 text-center">{error}</p>}
                <div className="flex justify-center mb-4">
                    <button
                        className={`px-4 py-2 mx-2 ${activeTab === 'create' ? 'bg-blue-500 text-white' : 'bg-gray-200'}`}
                        onClick={() => setActiveTab('create')}
                    >
                        Create Invoice
                    </button>
                    <button
                        className={`px-4 py-2 mx-2 ${activeTab === 'validate' ? 'bg-blue-500 text-white' : 'bg-gray-200'}`}
                        onClick={() => setActiveTab('validate')}
                    >
                        Validate Invoice
                    </button>
                    <button
                        className={`px-4 py-2 mx-2 ${activeTab === 'reference' ? 'bg-blue-500 text-white' : 'bg-gray-200'}`}
                        onClick={() => setActiveTab('reference')}
                    >
                        Reference Data
                    </button>
                    <button
                        className={`px-4 py-2 mx-2 ${activeTab === 'settings' ? 'bg-blue-500 text-white' : 'bg-gray-200'}`}
                        onClick={() => setActiveTab('settings')}
                    >
                        Settings
                    </button>
                </div>

                {activeTab === 'create' && (
                    <div className="bg-white p-6 rounded shadow max-w-4xl mx-auto">
                        <h2 className="text-2xl font-semibold mb-4">Create Invoice</h2>
                        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                            <div>
                                <label className="block text-sm font-medium">Invoice Type</label>
                                <select name="invoiceType" value={invoiceData.invoiceType} onChange={handleInputChange} className="w-full p-2 border rounded">
                                    {docTypes.map(doc => (
                                        <option key={doc.docTypeId} value={doc.docDescription}>{doc.docDescription}</option>
                                    ))}
                                </select>
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Invoice Date</label>
                                <input type="date" name="invoiceDate" value={invoiceData.invoiceDate} onChange={handleInputChange} className="w-full p-2 border rounded" required />
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Seller NTN/CNIC</label>
                                <input type="text" name="sellerNTNCNIC" value={invoiceData.sellerNTNCNIC} onChange={handleInputChange} className="w-full p-2 border rounded" required />
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Seller Business Name</label>
                                <input type="text" name="sellerBusinessName" value={invoiceData.sellerBusinessName} onChange={handleInputChange} className="w-full p-2 border rounded" required />
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Seller Province</label>
                                <select name="sellerProvince" value={invoiceData.sellerProvince} onChange={handleInputChange} className="w-full p-2 border rounded">
                                    <option value="">Select Province</option>
                                    {provinces.map(prov => (
                                        <option key={prov.stateProvinceCode} value={prov.stateProvinceDesc}>{prov.stateProvinceDesc}</option>
                                    ))}
                                </select>
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Seller Address</label>
                                <input type="text" name="sellerAddress" value={invoiceData.sellerAddress} onChange={handleInputChange} className="w-full p-2 border rounded" required />
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Buyer NTN/CNIC</label>
                                <input type="text" name="buyerNTNCNIC" value={invoiceData.buyerNTNCNIC} onChange={handleInputChange} className="w-full p-2 border rounded" />
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Buyer Business Name</label>
                                <input type="text" name="buyerBusinessName" value={invoiceData.buyerBusinessName} onChange={handleInputChange} className="w-full p-2 border rounded" required />
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Buyer Province</label>
                                <select name="buyerProvince" value={invoiceData.buyerProvince} onChange={handleInputChange} className="w-full p-2 border rounded">
                                    <option value="">Select Province</option>
                                    {provinces.map(prov => (
                                        <option key={prov.stateProvinceCode} value={prov.stateProvinceDesc}>{prov.stateProvinceDesc}</option>
                                    ))}
                                </select>
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Buyer Address</label>
                                <input type="text" name="buyerAddress" value={invoiceData.buyerAddress} onChange={handleInputChange} className="w-full p-2 border rounded" required />
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Buyer Registration Type</label>
                                <select name="buyerRegistrationType" value={invoiceData.buyerRegistrationType} onChange={handleInputChange} className="w-full p-2 border rounded">
                                    <option value="Registered">Registered</option>
                                    <option value="Unregistered">Unregistered</option>
                                </select>
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Invoice Reference No</label>
                                <input type="text" name="invoiceRefNo" value={invoiceData.invoiceRefNo} onChange={handleInputChange} className="w-full p-2 border rounded" />
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Scenario ID</label>
                                <input type="text" name="scenarioId" value={invoiceData.scenarioId} onChange={handleInputChange} className="w-full p-2 border rounded" required />
                            </div>
                        </div>
                        <h3 className="text-xl font-semibold mt-6 mb-4">Items</h3>
                        {invoiceData.items.map((item, index) => (
                            <div key={index} className="border p-4 mb-4 rounded">
                                <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                                    <div>
                                        <label className="block text-sm font-medium">HS Code</label>
                                        <input type="text" value={item.hsCode} onChange={(e) => handleInputChange(e, index, 'hsCode')} className="w-full p-2 border rounded" required />
                                    </div>
                                    <div>
                                        <label className="block text-sm font-medium">Product Description</label>
                                        <input type="text" value={item.productDescription} onChange={(e) => handleInputChange(e, index, 'productDescription')} className="w-full p-2 border rounded" required />
                                    </div>
                                    <div>
                                        <label className="block text-sm font-medium">Rate</label>
                                        <input type="text" value={item.rate} onChange={(e) => handleInputChange(e, index, 'rate')} className="w-full p-2 border rounded" required />
                                    </div>
                                    <div>
                                        <label className="block text-sm font-medium">Unit of Measurement</label>
                                        <input type="text" value={item.uoM} onChange={(e) => handleInputChange(e, index, 'uoM')} className="w-full p-2 border rounded" required />
                                    </div>
                                    <div>
                                        <label className="block text-sm font-medium">Quantity</label>
                                        <input type="number" value={item.quantity} onChange={(e) => handleInputChange(e, index, 'quantity')} className="w-full p-2 border rounded" required />
                                    </div>
                                    <div>
                                        <label className="block text-sm font-medium">Value Sales Excluding ST</label>
                                        <input type="number" value={item.valuesalesExcludingST} onChange={(e) => handleInputChange(e, index, 'valuesalesExcludingST')} className="w-full p-2 border rounded" required />
                                    </div>
                                    <div>
                                        <label className="block text-sm font-medium">Sales Tax Applicable</label>
                                        <input type="number" value={item.salesTaxApplicable} onChange={(e) => handleInputChange(e, index, 'salesTaxApplicable')} className="w-full p-2 border rounded" required />
                                    </div>
                                    <div>
                                        <label className="block text-sm font-medium">Further Tax</label>
                                        <input type="number" value={item.furtherTax} onChange={(e) => handleInputChange(e, index, 'furtherTax')} className="w-full p-2 border rounded" />
                                    </div>
                                </div>
                            </div>
                        ))}
                        <button onClick={addItem} className="bg-green-500 text-white px-4 py-2 rounded mt-4">Add Item</button>
                        <div className="mt-6 flex justify-end">
                            <button onClick={() => handleSubmit('https://gw.fbr.gov.pk/di data/v1/di/postinvoicedata')} className="bg-blue-500 text-white px-4 py-2 rounded">Submit Invoice</button>
                        </div>
                        <div className="mt-4 text-center">
                            <p className="text-sm">Ensure QR Code (Version 2.0, 1.0x1.0 inch) and FBR Digital Invoicing System logo are printed on the invoice.</p>
                        </div>
                    </div>
                )}

                {activeTab === 'validate' && (
                    <div className="bg-white p-6 rounded shadow max-w-4xl mx-auto">
                        <h2 className="text-2xl font-semibold mb-4">Validate Invoice</h2>
                        <p className="mb-4">Enter invoice data as in Create Invoice tab and submit for validation.</p>
                        <button onClick={() => handleSubmit('https://gw.fbr.gov.pk/di data/v1/di/validateinvoicedata')} className="bg-blue-500 text-white px-4 py-2 rounded">Validate Invoice</button>
                    </div>
                )}

                {activeTab === 'reference' && (
                    <div className="bg-white p-6 rounded shadow max-w-4xl mx-auto">
                        <h2 className="text-2xl font-semibold mb-4">Reference Data</h2>
                        <div className="space-y-4">
                            <div>
                                <h3 className="text-lg font-medium">Provinces</h3>
                                <ul className="list-disc pl-5">
                                    {provinces.map(prov => (
                                        <li key={prov.stateProvinceCode}>{prov.stateProvinceDesc} (Code: {prov.stateProvinceCode})</li>
                                    ))}
                                </ul>
                            </div>
                            <div>
                                <h3 className="text-lg font-medium">Document Types</h3>
                                <ul className="list-disc pl-5">
                                    {docTypes.map(doc => (
                                        <li key={doc.docTypeId}>{doc.docDescription} (ID: {doc.docTypeId})</li>
                                    ))}
                                </ul>
                            </div>
                            <div>
                                <h3 className="text-lg font-medium">Other Reference APIs</h3>
                                <p>Access HS Codes, SRO Items, UOMs, etc., via respective APIs listed in the documentation.</p>
                            </div>
                        </div>
                    </div>
                )}

                {activeTab === 'settings' && (
                    <div className="bg-white p-6 rounded shadow max-w-4xl mx-auto">
                        <h2 className="text-2xl font-semibold mb-4">Settings</h2>
                        <div className="space-y-4">
                            <div>
                                <label className="block text-sm font-medium">API Security Token</label>
                                <input type="text" placeholder="Enter your FBR-issued token" className="w-full p-2 border rounded" />
                            </div>
                            <div>
                                <label className="block text-sm font-medium">Environment</label>
                                <select className="w-full p-2 border rounded">
                                    <option value="sandbox">Sandbox</option>
                                    <option value="production">Production</option>
                                </select>
                            </div>
                        </div>
                    </div>
                )}
            </div>
        );
    }

    // Render the app
    ReactDOM.render(<App />, document.getElementById('root'));
</script>
